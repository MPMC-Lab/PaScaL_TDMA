46  double precision :: viscous_e1, viscous_e2, viscous_e3, viscous ! Viscous terms

58  ! Calculating r.h.s.
59  allocate( rhs(0:nx_sub, 0:ny_sub, 0:nz_sub))
60  do k = 1, nz_sub-1
61  kp = k+1
62  km = k-1
63 
64  do j = 1, ny_sub-1
65  jp = j + 1
66  jm = j - 1
67  jep = jpbc_index(j)
68  jem = jmbc_index(j)
69 
70  do i = 1, nx_sub-1
71  ip = i+1
72  im = i-1
73 
74  ! Diffusion term
75  dedx1 = ( theta(i ,j ,k ) - theta(im,j ,k ) )/dmx_sub(i )
76  dedx2 = ( theta(ip,j ,k ) - theta(i ,j ,k ) )/dmx_sub(ip)
77  dedy3 = ( theta(i ,j ,k ) - theta(i ,jm,k ) )/dmy_sub(j )
78  dedy4 = ( theta(i ,jp,k ) - theta(i ,j ,k ) )/dmy_sub(jp)
79  dedz5 = ( theta(i ,j ,k ) - theta(i ,j ,km) )/dmz_sub(k )
80  dedz6 = ( theta(i ,j ,kp) - theta(i ,j ,k ) )/dmz_sub(kp)
81 
82  viscous_e1 = 1.d0/dx*(dedx2 - dedx1)
83  viscous_e2 = 1.d0/dy*(dedy4 - dedy3)
84  viscous_e3 = 1.d0/dz*(dedz6 - dedz5)
85  viscous = 0.5d0*ct*(viscous_e1 + viscous_e2 + viscous_e3)
86 
87  ! Boundary treatment for the y-direction only
88  ebc_down = 0.5d0*ct/dy/dmy_sub(j)*thetabc3_sub(i,k)
89  ebc_up = 0.5d0*ct/dy/dmy_sub(jp)*thetabc4_sub(i,k)
90  ebc = dble(1. - jem)*ebc_down + dble(1. - jep)*ebc_up
91 
92  ! Diffusion term from incremental notation in next time step: x-direction
93  eapi = -0.5d0*ct/dx/dmx_sub(ip)
94  eami = -0.5d0*ct/dx/dmx_sub(i )
95  eaci = 0.5d0*ct/dx*( 1.d0/dmx_sub(ip) + 1.d0/dmx_sub(i) )
96 
97  ! Diffusion term from incremental notation in next time step: z-direction
98  eapk = -0.5d0*ct/dz/dmz_sub(kp)
99  eamk = -0.5d0*ct/dz/dmz_sub(k )
100  eack = 0.5d0*ct/dz*( 1.d0/dmz_sub(kp) + 1.d0/dmz_sub(k) )
101 
102  ! Diffusion term from incremental notation in next time step: y-direction
103  eapj = -0.5d0*ct/dy*( 1.d0/dmy_sub(jp) )*dble(jep)
104  eamj = -0.5d0*ct/dy*( 1.d0/dmy_sub(j ) )*dble(jem)
105  eacj = 0.5d0*ct/dy*( 1.d0/dmy_sub(jp) + 1.d0/dmy_sub(j) )
106 
107  erhs = eapk*theta(i,j,kp) + eack*theta(i,j,k) + eamk*theta(i,j,km) &
108  & + eapj*theta(i,jp,k) + eacj*theta(i,j,k) + eamj*theta(i,jm,k) &
109  & + eapi*theta(ip,j,k) + eaci*theta(i,j,k) + eami*theta(im,j,k)
110 
111  ! r.h.s. term
112  rhs(i,j,k) = theta(i,j,k)/dt + viscous + ebc &
113  & - (theta(i,j,k)/dt + erhs)
114  enddo
115  enddo
116  enddo
117 
118  ! solve in the z-direction.
119  allocate( ap(1:nx_sub-1, 1:nz_sub-1), am(1:nx_sub-1, 1:nz_sub-1), ac(1:nx_sub-1, 1:nz_sub-1), ad(1:nx_sub-1, 1:nz_sub-1) )
120 
121  ! Create a PaScaL_TDMA plan for the tridiagonal systems.
122  call pascal_tdma_plan_many_create(pz_many, nx_sub-1, comm_1d_z%myrank, comm_1d_z%nprocs, comm_1d_z%mpi_comm)
123 
124  ! Build a coefficient matrix for the tridiagonal systems into a 2D array.
125  do j = 1, ny_sub-1
126  do k = 1, nz_sub-1
127  kp = k+1
128  do i = 1, nx_sub-1
129  ap(i,k) = -0.5d0*ct/dz/dmz_sub(kp)*dt
130  am(i,k) = -0.5d0*ct/dz/dmz_sub(k )*dt
131  ac(i,k) = 0.5d0*ct/dz*( 1.d0/dmz_sub(kp) + 1.d0/dmz_sub(k) )*dt + 1.d0
132 
133  ad(i,k) = rhs(i,j,k)*dt
134  enddo
135  enddo
136 
137  ! Solve the tridiagonal systems under the defined plan with periodic boundary conditions.
138  call pascal_tdma_many_solve_cycle(pz_many, am, ac, ap, ad,nx_sub-1,nz_sub-1)
139 
140  ! Return the solution to the r.h.s. plane-by-plane
141  do k = 1, nz_sub-1
142  rhs(1:nx_sub-1,j,k)=ad(1:nx_sub-1,k)
143  enddo
144  enddo
145 
146  ! Destroy the PaScaL_TDMA plan for the tridiagonal systems.
147  call pascal_tdma_plan_many_destroy(pz_many)
148  deallocate( ap, am, ac, ad )
149 
150  ! solve in the x-direction.
151  allocate( ap(1:nx_sub-1, 1:ny_sub-1), am(1:nx_sub-1, 1:ny_sub-1), ac(1:nx_sub-1, 1:ny_sub-1), ad(1:nx_sub-1, 1:ny_sub-1) )
152 
153  ! Create a PaScaL_TDMA plan for the tridiagonal systems.
154  call pascal_tdma_plan_many_create(py_many, nx_sub-1, comm_1d_y%myrank, comm_1d_y%nprocs, comm_1d_y%mpi_comm)
155 
156  ! Build a coefficient matrix for the tridiagonal systems into a 2D array.
157  do k = 1, nz_sub-1
158  do j = 1, ny_sub-1
159  jp = j + 1
160  jm = j - 1
161  jep = jpbc_index(j)
162  jem = jmbc_index(j)
163

178  enddo
179  end do
180  call pascal_tdma_plan_many_destroy(py_many)
181  deallocate( ap, am, ac, ad )

184  allocate( ap(1:ny_sub-1, 1:nx_sub-1), am(1:ny_sub-1, 1:nx_sub-1), ac(1:ny_sub-1, 1:nx_sub-1), ad(1:ny_sub-1, 1:nx_sub-1) )

187  call pascal_tdma_plan_many_create(px_many, ny_sub-1, comm_1d_x%myrank, comm_1d_x%nprocs, comm_1d_x%mpi_comm)
188

191  do j = 1, ny_sub-1
192  do i = 1, nx_sub-1
193  ip = i+1
194  im = i-1
195 
196  ap(j,i) = -0.5d0*ct/dx/dmx_sub(ip)*dt
197  am(j,i) = -0.5d0*ct/dx/dmx_sub(i )*dt
198  ac(j,i) = 0.5d0*ct/dx*( 1.d0/dmx_sub(ip) + 1.d0/dmx_sub(i) )*dt + 1.d0
199  ad(j,i) = rhs(i,j,k)
200  enddo
201  enddo
