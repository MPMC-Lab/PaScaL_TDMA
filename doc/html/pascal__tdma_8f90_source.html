310  d(1) = d(1)/b(1)
311  c(1) = c(1)/b(1)
312 
313  a(2) = a(2)/b(2)

319  d(i) = r*(d(i)-a(i)*d(i-1))
320  c(i) = r*c(i)
321  a(i) = -r*a(i)*a(i-1)
322  enddo
323 
324  ! Reduction step : elimination of upper diagonal elements
325  do i=n_row-2,2,-1
326  d(i) = d(i)-c(i)*d(i+1)
327  a(i) = a(i)-c(i)*a(i+1)
328  c(i) =-c(i)*c(i+1)
329  enddo
330 
331  r = 1.d0/(1.d0-a(2)*c(1))
332  d(1) = r*(d(1)-c(1)*d(2))
333  a(1) = r*a(1)
334  c(1) = -r*c(1)*c(2)
335

347  plan%B_rt, 2, mpi_double_precision, &
348  plan%gather_rank, plan%ptdma_world, request(2), ierr)
349  call mpi_igather(plan%C_rd, 2, mpi_double_precision, &

353  plan%D_rt, 2, mpi_double_precision, &
354  plan%gather_rank, plan%ptdma_world, request(4), ierr)
355  call mpi_waitall(4, request, mpi_statuses_ignore, ierr)
356 
357  ! Solve the reduced tridiagonal system on plan%gather_rank
358  if(plan%myrank == plan%gather_rank) then
359  call tdma_single(plan%A_rt,plan%B_rt,plan%C_rt,plan%D_rt, plan%n_row_rt)
360  endif
361 
362  ! Scatter the solutions to each rank
363  call mpi_iscatter(plan%D_rt, 2, mpi_double_precision, &

365  plan%gather_rank, plan%ptdma_world, request(1), ierr)
366 
367  call mpi_waitall(1, request, mpi_statuses_ignore, ierr)
368

370  d(1 ) = plan%D_rd(1)
371  d(n_row) = plan%D_rd(2)
372  do i=2,n_row-1
373  d(i) = d(i)-a(i)*d(1)-c(i)*d(n_row)

484

494  ! The modified Thomas algorithm : elimination of lower diagonal elements.
495  ! First index is to indicate a number of independent many tridiagonal systems to use vectorization
496  ! Second index is to indicate a row number in a partitioned tridiagonal system
497  do i=1, n_sys
498  a(i,1) = a(i,1)/b(i,1)
499  d(i,1) = d(i,1)/b(i,1)
500  c(i,1) = c(i,1)/b(i,1)
501

503  d(i,2) = d(i,2)/b(i,2)
504  c(i,2) = c(i,2)/b(i,2)
505  enddo
506 
507  do j=3, n_row
508  do i=1, n_sys
509  r = 1.d0/(b(i,j)-a(i,j)*c(i,j-1))
510  d(i,j) = r*(d(i,j)-a(i,j)*d(i,j-1))
511  c(i,j) = r*c(i,j)
512  a(i,j) = -r*a(i,j)*a(i,j-1)
513  enddo
514  enddo
515 
516  ! The modified Thomas algorithm : elimination of upper diagonal elements
517  do j=n_row-2, 2, -1
518  do i=1, n_sys
519  d(i,j) = d(i,j)-c(i,j)*d(i,j+1)
520  a(i,j) = a(i,j)-c(i,j)*a(i,j+1)
521  c(i,j) =-c(i,j)*c(i,j+1)
522  enddo
523  enddo
524 
525  do i=1, n_sys
526  r = 1.d0/(1.d0-a(i,2)*c(i,1))
527  d(i,1) = r*(d(i,1)-c(i,1)*d(i,2))
528  a(i,1) = r*a(i,1)
529  c(i,1) = -r*c(i,1)*c(i,2)
530 
531  ! Construct many reduced tridiagonal systems per each rank. Each process has two rows of reduced systems.
532  plan%A_rd(i,1) = a(i,1); plan%A_rd(i,2) = a(i,n_row)
533  plan%B_rd(i,1) = 1.d0 ; plan%B_rd(i,2) = 1.d0
534  plan%C_rd(i,1) = c(i,1); plan%C_rd(i,2) = c(i,n_row)
535  plan%D_rd(i,1) = d(i,1); plan%D_rd(i,2) = d(i,n_row)
536  enddo
537 
538 
539  ! Transpose the reduced systems of equations for TDMA using MPI_Ialltoallw and derived datatypes
540  call mpi_ialltoallw(plan%A_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &

547  plan%C_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
548  plan%ptdma_world, request(3), ierr)
549  call mpi_ialltoallw(plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
550  plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
551  plan%ptdma_world, request(4), ierr)
552 
553  call mpi_waitall(4, request, mpi_statuses_ignore, ierr)
554

556  call tdma_many(plan%A_rt,plan%B_rt,plan%C_rt,plan%D_rt, plan%n_sys_rt, plan%n_row_rt)
557 
558  ! Transpose the obtained solutions to original reduced forms using MPI_Ialltoallw and derived datatypes
559  call mpi_ialltoallw(plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
560  plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
561  plan%ptdma_world, request(1), ierr)
562  call mpi_waitall(1, request, mpi_statuses_ignore, ierr)
563

565  do i=1,n_sys
566  d(i,1 ) = plan%D_rd(i,1)
567  d(i,n_row) = plan%D_rd(i,2)
568  enddo
569

571  do i=1,n_sys
572  d(i,j) = d(i,j)-a(i,j)*d(i,1)-c(i,j)*d(i,n_row)
573  enddo
574  enddo

665  call mpi_ialltoallw(plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
666  plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
667  plan%ptdma_world, request(1), ierr)
