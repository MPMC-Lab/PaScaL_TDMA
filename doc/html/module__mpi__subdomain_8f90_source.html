277  subroutine module_mpi_subdomain_mesh(myrank_in_x, myrank_in_y, myrank_in_z, &
278  nprocs_in_x, nprocs_in_y, nprocs_in_z)
279 
280  implicit none
281  integer, intent(in) :: myrank_in_x,myrank_in_y,myrank_in_z,nprocs_in_x,nprocs_in_y,nprocs_in_z
282 
283  integer :: i,j,k
284 
285  ! X-DIRECTION, x_sub are coordinates and dmx_sub are grid lengths
286  dx = lx/dble(nx-1)
287  do i = 0, nx_sub
288  x_sub(i) = dble(i-1+ista-1)*dx
289  dmx_sub(i)=dx
290  end do
291 
292  ! Y-DIRECTION, y_sub are coordinates and dmy_sub are grid lengths
293  dy = ly/dble(ny-1)
294  do j = 0, ny_sub
295  y_sub(j) = dble(j-1+jsta-1)*dy
296  dmy_sub(j)=dy
297  end do
298 
299  ! Z-DIRECTION, z_sub are coordinates and dmz_sub are grid lengths
300  dz = lz/dble(nz-1)
301  do k = 0, nz_sub
302  z_sub(k) = dble(k-1+ksta-1)*dz
303  dmz_sub(k)=dz
304  end do
305 
306  ! For boundary grids in x-direction. Same with other grids due to p.b.c.
307  if(myrank_in_x == 0) then
308  dmx_sub(0)= dx
309  else if(myrank_in_x == nprocs_in_x-1) then
310  dmx_sub(nx_sub)=dx
311  endif
312 
313  ! For boundary grids in y-direction. Half grid length for lower and upper boundary grids.
314  if(myrank_in_y == 0) then
315  y_sub(0) = 0.0d0
316  dmy_sub(0)=0.0d0
317  dmy_sub(1)=dy/2.0d0
318  else if(myrank_in_y == nprocs_in_y-1) then
319  dmy_sub(ny_sub)=dy/2.0d0
320  endif
321 
322  ! For boundary grids in z-direction. Same with other grids due to p.b.c.
323  if(myrank_in_z == 0) then
324  dmz_sub(0)=dz
325  else if(myrank_in_z == nprocs_in_z-1) then
326  dmz_sub(nz_sub)=dz
327  endif
328 
329  return

339  subroutine module_mpi_subdomain_initialization(theta_sub, myrank_in_y, nprocs_in_y)
340  implicit none
341  integer, intent(in) :: myrank_in_y, nprocs_in_y
342  double precision, intent(inout) :: theta_sub(0:nx_sub, 0:ny_sub, 0:nz_sub)
343 
344  double precision, parameter :: pi = acos(-1.d0)
345  integer :: i,j,k
346 
347  ! Initialize the main variable with a sine function and linearly changed values between wall boundaries
348  do k = 0, nz_sub
349  do j = 0, ny_sub
350  do i = 0, nx_sub
351  theta_sub(i,j,k) = (theta_cold-theta_hot)/ly*y_sub(j)+theta_hot &
352  & + sin(2*2.d0*pi/lx *x_sub(i)) &
353  & * sin(2*2.d0*pi/lz *z_sub(k)) &
354  & * sin(2*2.d0*pi/ly *y_sub(j))
355  end do
356  end do
357  end do
358 
359  ! Initialize the upper and lower boundary grids
360  do k = 0, nz_sub
361  do i = 0, nx_sub
362  if(myrank_in_y==0) theta_sub(i,0,k) = theta_hot
363  if(myrank_in_y==nprocs_in_y-1) theta_sub(i,ny_sub,k) = theta_cold
364  end do
365  end do
366 
367  return
368  end subroutine module_mpi_subdomain_initialization
